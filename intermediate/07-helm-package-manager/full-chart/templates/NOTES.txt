{{- if eq .Values.service.type "LoadBalancer" }}
  Get the application URL by running:
    kubectl get service {{ include "full-webapp.fullname" . }} --namespace {{ .Release.Namespace }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}'

{{- else if eq .Values.service.type "NodePort" }}
  Get the application URL by running:
    export NODE_PORT=$(kubectl get service {{ include "full-webapp.fullname" . }} --namespace {{ .Release.Namespace }} -o jsonpath='{.spec.ports[0].nodePort}')
    export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath='{.items[0].status.addresses[0].address}')
    echo "Application URL: http://$NODE_IP:$NODE_PORT"

{{- else }}
  The application is available via ClusterIP. Forward the port:
    kubectl port-forward service/{{ include "full-webapp.fullname" . }} 8080:{{ .Values.service.port }} --namespace {{ .Release.Namespace }}
    echo "Application URL: http://localhost:8080"
{{- end }}
